<script>
  let stories = JSON.parse(localStorage.getItem('stories')) || [];

  function saveStories() {
    localStorage.setItem('stories', JSON.stringify(stories));
  }

  function publishStory() {
    const author = document.getElementById('author').value.trim();
    const title = document.getElementById('title').value.trim();
    const content = document.getElementById('content').value.trim();

    if (!author || !title || !content) {
      alert('Please fill out all fields.');
      return;
    }

    const story = {
      id: Date.now(),
      author,
      title,
      content,
      likes: 0,
      date: new Date().toLocaleString()
    };

    stories.unshift(story);
    saveStories();
    renderStories();

    document.getElementById('author').value = '';
    document.getElementById('title').value = '';
    document.getElementById('content').value = '';
  }

  function likeStory(id) {
    const story = stories.find(s => s.id === id);
    story.likes++;
    saveStories();
    renderStories();
  }

  function deleteStory(id) {
    if (!confirm('Delete this story?')) return;
    stories = stories.filter(s => s.id !== id);
    saveStories();
    renderStories();
  }

  function renderStories() {
    const list = document.getElementById('storyList');
    list.innerHTML = '';

    if (stories.length === 0) {
      list.innerHTML = '<p>No stories yet. Be the first to publish!</p>';
      return;
    }

    stories.forEach(story => {
      const div = document.createElement('div');
      div.className = 'card story';
      div.innerHTML = `
        <h3>${story.title}</h3>
        <small>by ${story.author} ¬∑ ${story.date}</small>
        <p>${story.content}</p>
        <div class="actions">
          <button onclick="likeStory(${story.id})">‚ù§Ô∏è ${story.likes}</button>
          <button onclick="deleteStory(${story.id})">üóëÔ∏è Delete</button>
        </div>
      `;
      list.appendChild(div);
    });
  }

  renderStories();
</script>
